-- DRAGGING

local dragging = false
local dragInput = nil
local dragStart = nil
local startPos = nil
local dragMoved = false

local function updateDrag(input)
    local delta = input.Position - dragStart
    local screenSize = mainGui.AbsoluteSize
    local frameSize = dragFrame.AbsoluteSize
    local absX = math.clamp(startPos.X.Offset + delta.X, 0, screenSize.X - frameSize.X)
    local absY = math.clamp(startPos.Y.Offset + delta.Y, 0, screenSize.Y - frameSize.Y)
    dragFrame.Position = UDim2.new(0, absX, 0, absY)
end

button.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1
        or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragMoved = false
        dragInput = input
        dragStart = input.Position
        startPos = dragFrame.Position
    end
end)

button.InputChanged:Connect(function(input)
    if input == dragInput then
        if (input.Position - dragStart).Magnitude > 6 then dragMoved = true end
        if dragMoved then updateDrag(input) end
    end
end)

button.InputEnded:Connect(function(input)
    if input == dragInput then
        dragging = false
        dragInput = nil
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if dragging and input == dragInput and dragMoved then updateDrag(input) end
end)

local function getLocalRoot()
    local char = player.Character
    if not char then return nil end
    return char:FindFirstChild("HumanoidRootPart")
end

local function isTargetValid(target)
    if not target then return false end
    if not target.Parent then return false end
    if target.Health <= 0 then return false end
    local root = target.Parent:FindFirstChild("HumanoidRootPart")
    return root ~= nil
end

local function findNearestTarget()
    local myRoot = getLocalRoot()
    if not myRoot then return nil, nil end

    local myPos = myRoot.Position
    local lookDir = camera.CFrame.LookVector
    lookDir = Vector3.new(lookDir.X, 0, lookDir.Z).Unit

    local bestTarget = nil
    local bestRoot = nil
    local bestScore = math.huge

    for _, p in ipairs(Players:GetPlayers()) do
        if p == player then continue end
        local char = p.Character
        if not char then continue end
        local humanoid = char:FindFirstChildOfClass("Humanoid")
        local root = char:FindFirstChild("HumanoidRootPart")
        if not humanoid or not root then continue end
        if humanoid.Health <= 0 then continue end

        local toTarget = root.Position - myPos
        local dist = toTarget.Magnitude
        if dist > CONFIG.MAX_DISTANCE then continue end

        local toTargetFlat = Vector3.new(toTarget.X, 0, toTarget.Z)
        if toTargetFlat.Magnitude < 0.001 then continue end
        toTargetFlat = toTargetFlat.Unit

        local dot = lookDir:Dot(toTargetFlat)
        local angle = math.deg(math.acos(math.clamp(dot, -1, 1)))
        if angle > CONFIG.DETECTION_ANGLE then continue end

        local score = dist + (angle * 1.5)
        if score < bestScore then
            bestScore = score
            bestTarget = humanoid
            bestRoot = root
        end
    end

    -- Also search workspace for NPCs/dummies with humanoids
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("Humanoid") and obj.Health > 0 then
            local model = obj.Parent
            if model == player.Character then continue end
            local root = model:FindFirstChild("HumanoidRootPart")
            if not root then continue end

            local toTarget = root.Position - myPos
            local dist = toTarget.Magnitude
            if dist > CONFIG.MAX_DISTANCE then continue end

            local toTargetFlat = Vector3.new(toTarget.X, 0, toTarget.Z)
            if toTargetFlat.Magnitude < 0.001 then continue end
            toTargetFlat = toTargetFlat.Unit

            local dot = lookDir:Dot(toTargetFlat)
            local angle = math.deg(math.acos(math.clamp(dot, -1, 1)))
            if angle > CONFIG.DETECTION_ANGLE then continue end

            local score = dist + (angle * 1.5)
            if score < bestScore then
                bestScore = score
                bestTarget = obj
                bestRoot = root
            end
        end
    end

    return bestTarget, bestRoot
end

local function disableLockOn()
    isLockedOn = false
    lockedTarget = nil
    lockedTargetRoot = nil
    camera.CameraType = Enum.CameraType.Custom
    idlePulsing = true
    TweenService:Create(glowFrame, TweenInfo.new(0.3), { BackgroundTransparency = 0.82 }):Play()
    setUI("wawalock v2", Color3.fromRGB(255, 80, 80), "press to lock nearest target", Color3.fromRGB(200, 200, 200))
end

local function enableLockOn()
    local target, root = findNearestTarget()
    if not target or not root then
        -- Flash yellow: no target
        setUI("no target!", Color3.fromRGB(255, 220, 0), "no humanoid in range", Color3.fromRGB(255, 220, 0))
        task.delay(1.2, function()
            if not isLockedOn then
                setUI("wawalock v2", Color3.fromRGB(255, 80, 80), "press to lock nearest target", Color3.fromRGB(200, 200, 200))
            end
        end)
        return false
    end

    isLockedOn = true
    lockedTarget = target
    lockedTargetRoot = root
    idlePulsing = false
    currentCamCFrame = camera.CFrame

    camera.CameraType = Enum.CameraType.Scriptable
    TweenService:Create(glowFrame, TweenInfo.new(0.3), { BackgroundTransparency = 0.65 }):Play()
    TweenService:Create(neonStroke, TweenInfo.new(0.3), { Transparency = 0 }):Play()
    setUI("🔴 wawalock v2", Color3.fromRGB(255, 100, 100), "locked: " .. (target.Parent and target.Parent.Name or "?"), Color3.fromRGB(255, 60, 60))

    return true
end

-- Camera update loop (your uploaded system)
RunService.Heartbeat:Connect(function()
    if not isLockedOn then return end

    if not isTargetValid(lockedTarget) then
        disableLockOn()
        return
    end

    local myRoot = getLocalRoot()
    if not myRoot then disableLockOn() return end

    local targetPos = lockedTargetRoot.Position + Vector3.new(0, CONFIG.TARGET_HEIGHT_OFFSET, 0)
    local myPos = myRoot.Position

    -- Rotate character to face target
    local toTarget = targetPos - myPos
    local toTargetFlat = Vector3.new(toTarget.X, 0, toTarget.Z)
    if toTargetFlat.Magnitude > 0.1 then
        myRoot.CFrame = myRoot.CFrame:Lerp(
            CFrame.new(myPos) * CFrame.Angles(0, math.atan2(-toTargetFlat.X, -toTargetFlat.Z), 0),
            0.2
        )
    end

    -- Camera behind player looking at target
    local pitchRad = math.rad(CONFIG.CAMERA_PITCH_OFFSET)
    local dirToTarget = toTargetFlat.Magnitude > 0.1 and toTargetFlat.Unit or Vector3.new(0, 0, -1)
    local camBack = -dirToTarget
    local camOffset = (camBack * CONFIG.CAMERA_DISTANCE) + Vector3.new(0, CONFIG.CAMERA_DISTANCE * math.tan(pitchRad), 0)
    local desiredCamPos = myPos + camOffset + Vector3.new(0, 1.5, 0)
    local desiredCF = CFrame.lookAt(desiredCamPos, targetPos)

    currentCamCFrame = currentCamCFrame:Lerp(desiredCF, CONFIG.CAMERA_LERP_SPEED)
    camera.CFrame = currentCamCFrame
end)
